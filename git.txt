#------------------------------------------------
# apprendre git
#------------------------------------------------
# installer git sous debian
#------------------------------------------------
sudo apt install git-all
#------------------------------------------------
# cloner un depot
#------------------------------------------------
git clone {url-depot-https}
#------------------------------------------------
# cloner un depot avec un nom local
#------------------------------------------------
git clone {url-depot-https} {nom-depot-local}
#------------------------------------------------
# afficher l'etat d'un depot
#------------------------------------------------
git status
#------------------------------------------------
# afficher l'etat d'un depot
# afficher les chemins des fichies
#------------------------------------------------
git status -u
#------------------------------------------------
# indexer tous les fichiers
#------------------------------------------------
git add .
#------------------------------------------------
# reindexer tous les fichiers
#------------------------------------------------
git add --renormalize .
#------------------------------------------------
# afficher une configuration {config}
#------------------------------------------------
git config --get {config}
#------------------------------------------------
# afficher toutes les configurations
#------------------------------------------------
git config --list
#------------------------------------------------
# afficher les caracteres de fin de ligne des fichiers
#------------------------------------------------
git ls-files --eol
#------------------------------------------------
# configurer le caractere de fin de ligne des fichiers
#------------------------------------------------
.gitattributes
...
{pattern} text eol=crlf
